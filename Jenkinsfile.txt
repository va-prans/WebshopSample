node('docker') {
    stage 'Checkout'
        checkout([$class: 'GitSCM', branches: [[name: '*/master']], doGenerateSubmoduleConfigurations: false, extensions: [], submoduleCfg: [], userRemoteConfigs: [[url: 'https://github.com/va-prans/WebshopSample']]])
    stage 'Run Compose Down'
        sh "docker-compose -f docker-compose.yml down"
	stage 'Run Compose Up'
        sh "docker-compose -f docker-compose.yml up --force-recreate --abort-on-container-exit"
}
